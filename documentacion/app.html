115            if hilo_guardar.is_alive():

221        """
222        if not self._curp_seleccionado:
223            messagebox.showwarning('Seleccione un paciente', 'Seleccione un paciente de la tabla.')

226        nombre = self._entradas['nombre'].get().strip()
227        curp_nuevo = self._entradas['curp'].get().strip()
228        sexo = self._entradas['sexo'].get().strip()
229        fecha = self._entradas['fecha'].get().strip()
230        if not nombre or not curp_nuevo:
231            messagebox.showwarning('Campos Vacíos', 'Seleccione un paciente de la tabla.')
232            return
233        
234        paciente_antiguo = self._repositorio.leer(self._curp_seleccionado)
235
236        if not paciente_antiguo:
237            messagebox.showwarning('Paciente No Encontrado', 'El paciente original no existe.')
238            return
239        
240        if curp_nuevo != self._curp_seleccionado:
241            if self._repositorio.leer(curp_nuevo):
242                messagebox.showwarning('Paciente Ya Existe', 'El paciente con el curp seleccionado ya existe.')
243                return
244        try:
245            cadena_csv = f"{nombre},{curp_nuevo},{sexo},{fecha}"
246            paciente_actualizado = PacientePediatrico.desde_cadena(cadena_csv)
247            paciente_actualizado.historial = paciente_antiguo.historial
248            paciente_actualizado.historial_consultas = paciente_antiguo.historial_consultas
249
250            if curp_nuevo != self._curp_seleccionado:
251                self._repositorio.eliminar(self._curp_seleccionado)
252                exito = self._repositorio.crear(paciente_actualizado)
253            else:
254                exito = self._repositorio.actualizar(paciente_actualizado)
255            
256            if exito:
257                messagebox.showinfo('Éxito', 'Paciente actualizado exitosamente.')
258                self._limpiar_formulario()
259                self._actualizar_tabla()
260            else:
261                messagebox.showerror('Error', 'Error al actualizar el paciente.')
262        except ValueError as ve:
263            messagebox.showwarning("Validación", f'Datos Incorrectos: {ve}')
264        except Exception as e:
265            messagebox.showerror('Error. Está mal en algo.', str(e))
