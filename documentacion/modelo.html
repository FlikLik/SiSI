  9class Persona(ABC):
 10    """
 11    Clase abstracta base que encapsula la identidad general de la persona dentro del sistema.
 12    """
 13
 14    def __init__(self, nombre: str, sexo: str, curp: str):
 15        self._nombre = nombre
 16        self._sexo = sexo
 17        self.curp = curp
 18    """
 19    Incializador de la clase 'Modelo' que recibe los datos generales de un individio cualquiera.
 20    :param nombre: Nombre completo de la persona
 21    :param sexo: Sexo de la persona
 22    :param curp: Curp de la persona
 23    """
 24
 25    @property
 26    def curp(self) -> str:
 27        """
 28        Método que define el curp de la persona como propiedad.
 29        """
 30        return self._curp
 31    
 32    @curp.setter
 33    def curp(self, curp: str):
 34        self._curp = curp.strip().upper() if curp else ''
 35        if not self.verificar_curp():
 36            raise ValueError(f"El CURP '{curp}' no tiene un formato válido.")
 37    """
 38    Método que asigna el curp a la propiedad si al evaluar el curp es correcto. Si es así, lo asigna a la propiedad, en caso contrario manda un mensaje de error.
 39    """
 40
 41    def verificar_curp(self) -> bool:
 42        """
 43        Método que evalua si el curp cumple con el formato correcto para retornar un 'true' si se cumple, retorna un 'false' en caso contrario.
 44        """
 45        patron = r"^[A-Z]{4}\d{6}[HM][A-Z]{5}[A-Z\d]\d$"
 46        try:
 47            return bool(re.match(patron, self._curp))
 48        except TypeError:
 49            return False
 50
 51    def __str__(self):
 52        """
 53        Método que devuelve en forma de cadena de texto los elementos de la persona: nombre, sexo, curp.
 54        """
 55        return f"{self._nombre} {self._sexo} {self._curp}"

172        Método para calcular la edad del paciente en base al año de su fecha de nacimiento. Retorna la edad del paciente.

183        """
